    <!-- Script de interatividade exigido no teste -->
    <script>
      document.addEventListener("DOMContentLoaded", () => {
        // Slider Antes/Depois
        document.querySelectorAll('.service_before_after_slider').forEach(slider => {
          let active = false;
          let sc = slider.querySelector('.slider_before');
          let handle = slider.querySelector('.slider_handle');
          
          slider.addEventListener('mousedown', () => active = true);
          document.addEventListener('mouseup', () => active = false);
          document.addEventListener('mousemove', (e) => {
            if (!active) return;
            let x = e.pageX - slider.getBoundingClientRect().left;
            x = Math.max(0, Math.min(x, slider.offsetWidth));
            sc.style.width = x + 'px';
            handle.style.left = x + 'px';
          });
          slider.addEventListener('touchstart', () => active = true, {passive: true});
          document.addEventListener('touchend', () => active = false);
          document.addEventListener('touchmove', (e) => {
            if (!active) return;
            let x = e.touches[0].pageX - slider.getBoundingClientRect().left;
            x = Math.max(0, Math.min(x, slider.offsetWidth));
            sc.style.width = x + 'px';
            handle.style.left = x + 'px';
          }, {passive: true});
        });

        // Expandable Details
        document.querySelectorAll('.service_expandable_btn').forEach(btn => {
          btn.addEventListener('click', () => {
            const content = btn.nextElementSibling;
            const isExpanded = content.classList.contains('expanded');
            
            // Fecha todos os outros primeiro
            document.querySelectorAll('.service_expandable_content').forEach(c => c.classList.remove('expanded'));
            document.querySelectorAll('.service_expandable_btn').forEach(b => {
              b.setAttribute('aria-expanded', 'false');
              b.querySelector('svg').style.transform = 'rotate(0deg)';
            });

            if (!isExpanded) {
              content.classList.add('expanded');
              btn.setAttribute('aria-expanded', 'true');
              btn.querySelector('svg').style.transform = 'rotate(180deg)';
            }
          });
        });

        // Intersection Observer p/ Animações
        const observer = new IntersectionObserver((entries) => {
          entries.forEach(entry => {
            if (entry.isIntersecting) {
              entry.target.classList.add('is-visible');
            }
          });
        }, { threshold: 0.3 });

        document.querySelectorAll('.section_service_fullscreen').forEach(sec => {
          observer.observe(sec);
        });
        
      });
    </script>
